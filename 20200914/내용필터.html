//jQuery가 포함된 span태그 의 테두리 파랑 :
$('span:contains(jQuery)').css('border','2px solid blue');

//jQuery가 포함되지 않은 span 태그 의 테두리 빨강 :
$('span:not(:contains(jQuery))').css('border','3px solid red');

//span을 갖는 div요소 태그 의 테두리 그린
$('div:has(span)').css('border','2px solid green');

//아무것도 없는 빈 div 태그 의 테두리
$('div:empty').css('border','3px solid orange');

//div 후손 span 태그 의 글자색 빨강, 크기,굵게
$('#result1 div span').css({'color':'red','font-weight':'bold','font-size':'1.2em'});
